function sortBooks($books, $sortBy) {
  if ($sortBy == 'pages') {
    usort($books, function($a, $b) {
      return $a['pages'] <=> $b['pages'];
    });
  } else if ($sortBy == 'date') {
    usort($books, function($a, $b) {
      return strtotime($a['publish_date']) <=> strtotime($b['publish_date']);
    });
  }
  return $books;
}
